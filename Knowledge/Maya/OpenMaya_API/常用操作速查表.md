## 🔍 节点与对象查询

```python
# 获取选中对象
sel = om2.MGlobal.getActiveSelectionList()
dagPath = sel.getDagPath(0)

# 通过名称获取节点
sel = om2.MSelectionList()
sel.add("pCube1")
node = sel.getDependNode(0)  # MObject
dagPath = sel.getDagPath(0)  # MDagPath（需要是 DAG 节点）

# 获取节点名称
fn = om2.MFnDependencyNode(node)
print(fn.name())             # "pCube1"
print(fn.typeName)           # "mesh"

# 遍历场景中的所有节点
it = om2.MItDependencyNodes(om2.MFn.kMesh)
while not it.isDone():
    fn = om2.MFnDependencyNode(it.thisNode())
    print(fn.name())
    it.next()
```

## 🔌 Plug（属性）操作

```python
fn = om2.MFnDependencyNode(node)

# 获取 plug
plug = fn.findPlug("translateX", False)

# 读取值
value = plug.asFloat()     # asInt / asBool / asString / asMObject

# 设置值（需在 undo 队列中）
plug.setFloat(1.5)

# 连接 plug
src = fn1.findPlug("output", False)
dst = fn2.findPlug("input", False)
dgMod = om2.MDGModifier()

dgMod.doIt()
```

## 🔄 撤销块（Maya 红线规范）

```python
# 方式一：cmds undoInfo（快速脚本用）
cmds.undoInfo(openChunk=True, chunkName="myOperation")
try:
    # 你的操作
    pass
finally:
    cmds.undoInfo(closeChunk=True)

# 方式二：MDGModifier / MDagModifier（API操作首选）
dgMod = om2.MDGModifier()
dgMod.createNode("multiplyDivide")
dgMod.doIt()
# 撤销时：dgMod.undoIt()

# 方式三：自定义 MPxCommand（工具级）
class MyCmd(om2.MPxCommand):
    def doIt(self, args): ...
    def undoIt(self): ...
    def redoIt(self): ...
    def isUndoable(self): return True
```

## 🧩 自定义节点 MPxNode

```python
import maya.api.OpenMaya as om2

class MyNode(om2.MPxNode):
    kPluginNodeId = om2.MTypeId(0x00080000)
    
    # 定义属性
    inputAttr = om2.MObject()
    outputAttr = om2.MObject()
    
    @staticmethod
    def creator():
        return MyNode()
    
    @staticmethod
    def initialize():
        nAttr = om2.MFnNumericAttribute()
        MyNode.inputAttr = nAttr.create("input", "in", 
                                         om2.MFnNumericData.kFloat, 0.0)
        nAttr.writable = True
        MyNode.addAttribute(MyNode.inputAttr)
        
        MyNode.outputAttr = nAttr.create("output", "out",
                                          om2.MFnNumericData.kFloat, 0.0)
        nAttr.writable = False
        nAttr.storable = False
        MyNode.addAttribute(MyNode.outputAttr)
        
        MyNode.attributeAffects(MyNode.inputAttr, MyNode.outputAttr)
    
    def compute(self, plug, dataBlock):
        if plug == MyNode.outputAttr:
            inputVal = dataBlock.inputValue(MyNode.inputAttr).asFloat()
            outputHandle = dataBlock.outputValue(MyNode.outputAttr)
            outputHandle.setFloat(inputVal * 2.0)
            dataBlock.setClean(plug)

# 注册
def initializePlugin(plugin):
    pluginFn = om2.MFnPlugin(plugin)
    pluginFn.registerNode("myNode", MyNode.kPluginNodeId,
                           MyNode.creator, MyNode.initialize)

def uninitializePlugin(plugin):
    pluginFn = om2.MFnPlugin(plugin)
    pluginFn.deregisterNode(MyNode.kPluginNodeId)
```

## 💡 常见陷阱

| 问题 | 原因 | 解决 |
|:---|:---|:---|
| `MObject` 失效 | 场景操作后引用失效 | 每次操作前重新获取 |
| API 1.0 / 2.0 混用 | 返回类型不兼容 | 统一用 `maya.api.*` |
| 忘记 `doIt()` | MDGModifier 不执行 | 操作后必须调用 `doIt()` |
| 撤销失效 | 未包裹在 undoInfo 或 MPxCommand | 所有操作包裹撤销块 |
